*** Keywords ***
8zai install
    [Arguments]    ${app_name}
    Open Application    ${REMOTE_URL}    platformName=${platformName}    platformVersion=${platformVersion}    deviceName=${deviceName}    app=${app_path}${app_name}    appPackage=${8zai_appPackage}
    ...    appActivity=${8zai_appActivity_login}    unicodeKeyboard=True    resetKeyboard=True

8zai open
    Open Application    ${REMOTE_URL}    platformName=${platformName}    platformVersion=${platformVersion}    deviceName=${deviceName}    appPackage=${8zai_appPackage}    appActivity=${8zai_appActivity_login}
    Wait Until Keyword Succeeds    2min    5sec    Page Should Contain Element    id=com.ndol.sale.starter:id/edt_account

8zai login
    [Arguments]    ${contact_name}    ${contact_password}
    Get Network Connection Status    #获取设备连接状态
    Open Application    ${REMOTE_URL}    platformName=${platformName}    platformVersion=${platformVersion}    deviceName=${deviceName}    appPackage=${8dol_appPackage}    appActivity=${8dol_appActivity_login}
    ${date}    Get Str Date    #获取当前的日期作为子目录字符串
    Mkdir    ${8zai_image}\\${date}    #检查目录是否存在，没有就创建
    Wait Until Keyword Succeeds    2min    5sec    Page Should Contain Element    id=com.ndol.sale.starter:id/edt_account
    Capture Page Screenshot    ${8zai_image}\\${date}\\8dol_login1.png
    Input Text    name=com.ndol.sale.starter:id/edt_account    ${contact_name}
    Click Element    id=com.ndol.sale.starter:id/togbtn_show_pwd    #先点击一下显示明文密码，便于截图
    Click Element    id=com.ndol.sale.starter:id/edt_password    #先点击一下密码框，改变光标位置
    Go Back    #隐藏手机输入键盘
    Input PassWord    id=com.ndol.sale.starter:id/edt_password    ${contact_password}
    Capture Page Screenshot    ${8zai_image}\\${date}\\8dol_login2.png
    Click Element    id=com.ndol.sale.starter:id/btn_login
    Capture Page Screenshot    ${8zai_image}\\${date}\\8dol_login3.png
